 /** For seconds and minutes we have 60 increments so we should reotate the for each second and minute by 360/60=6**/

 /*** For hours we have 12 increments so 360/12=30deg. But the hour hand move with smaller amount based on the minute so once we have a hour and half the hand will be btween the prev and next hout that is why we should add + (mins/2)**/

 setInterval(function() {
     /*** Set the second rotation****/
     var seconds = new Date().getSeconds();
     var sdegree = seconds * 6;

     let secElement = document.querySelector("#sec");
     secElement.style.top = '50%';
     secElement.style.left = '50%';
     secElement.style.transform = 'translate(-50%, -50%) rotate(' + sdegree + 'deg)';

     /*** Set minute rotation ****/
     var minutes = new Date().getMinutes();
     var mdegree = minutes * 6;

     let minElement = document.querySelector("#min");
     minElement.style.top = '50%';
     minElement.style.left = '50%';
     minElement.style.transform = 'translate(-50%, -50%) rotate(' + mdegree + 'deg)';
     /*** Set Hour rotation ****/
     var hour = new Date().getHours();
     var hdegree = hour * 30 + (minutes / 2);

     let hourElement = document.querySelector("#hour");
     hourElement.style.top = '50%';
     hourElement.style.left = '50%';
     hourElement.style.transform = 'translate(-50%, -50%) rotate(' + hdegree + 'deg)';
 }, 1000);
